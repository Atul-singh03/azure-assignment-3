# ==== Basic Variables ====
$resourceGroup = "rg-powershell"
$location = "eastus"
$vmName = "myVM"

# ==== Create Resource Group ====
New-AzResourceGroup -Name $resourceGroup -Location $location

# ==== Create Credentials ====
$cred = Get-Credential -Message "Enter username as azureuser"

# ==== Create VM (GEN1, No SecurityProfile, Simple Config) ====
New-AzVM `
  -ResourceGroupName $resourceGroup `
  -Name $vmName `
  -Location $location `
  -ImageName "Win2016Datacenter" `
  -Credential $cred `
  -OpenPorts 3389 `
  -Size "Standard_B1s"
